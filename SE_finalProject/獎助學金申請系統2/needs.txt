
// === 獎助學金申請系統 EER → DBML（全中文命名版） ===

// 使用者（超類：學生、老師、系管、獎助單位共用的帳號）
Table 使用者 {
  帳號 varchar [primary key] // 帳號
  種類 varchar              // 種類：學生 / 老師 / 系統管理員 / 獎助單位
  姓名 varchar              // 姓名
  密碼 varchar              // 密碼
  手機 varchar              // 手機
  email varchar             // email
}

// 學生（次類：繼承 使用者.帳號）
Table 學生 {
  帳號 varchar [primary key, ref: > 使用者.帳號] // PK + FK → 使用者.帳號
  系所 varchar              // 系所
  性別 varchar              // 性別
  系級 varchar              // 系級
  班級 varchar              // 班級
  地址 varchar              // 地址
  申請紀錄 text             // 申請紀錄
}

// 老師（次類）
Table 老師 {
  帳號 varchar [primary key, ref: > 使用者.帳號] // PK + FK → 使用者.帳號
  科系 varchar              // 科系
}

// 系統管理員（次類）
Table 系統管理員 {
  帳號 varchar [primary key, ref: > 使用者.帳號] // PK + FK → 使用者.帳號
  處室 varchar              // 處室
}

// 獎助單位（次類）
Table 獎助單位 {
  帳號 varchar [primary key, ref: > 使用者.帳號] // PK + FK → 使用者.帳號
  名稱 varchar              // 名稱
  負責人 varchar            // 負責人
}

// 申請資料（Application）
Table 申請資料 {
  申請編號 integer [primary key]  // 申請編號
  申請日期 date                   // 申請日期
  自傳 text                      // 自傳

  學年 varchar                   // 學年
  學期 varchar                   // 學期
  申請項目 varchar               // 申請項目

  家庭居住狀況 varchar           // 家庭居住狀況
  個人居住狀況 varchar           // 個人居住狀況
  就學貸款 varchar               // 就學貸款（有/無 等）
  學雜費減免 varchar             // 學雜費減免
  上學期獲獎助學金名稱 varchar   // 上學期獲獎(助)學金名稱

  證明文件 text                  // 證明文件
  檢附申請文件 text              // 檢附申請文件
  家庭狀況說明 text              // 家庭狀況說明
  家庭成員狀況 text              // 家庭成員狀況
  與推薦人關係 varchar           // 與推薦人關係

 
  學生帳號 varchar [not null, ref: > 學生.帳號]        // 申請(學生–申請資料)
  獎助單位帳號 varchar [not null, ref: > 獎助單位.帳號]  // 審查(獎助單位–申請資料)
  獎學金編號 varchar [not null]                        // 與獎(助)學金組成複合 FK 的部分鍵

  通過與否 varchar              // 審查結果：通過與否
}

// 獎(助)學金（弱實體，依賴 獎助單位）
Table 獎助學金 {
  獎助單位帳號 varchar [not null, ref: > 獎助單位.帳號] // FK → 獎助單位.帳號
  獎學金編號 varchar [not null]        // 部分鍵（Partially Unique）
  名稱 varchar                         // 名稱
  金額 integer                         // 金額
  開始時間 date                        // 開始時間
  截止日期 date                        // 截止日期
  申請條件 text                        // 申請條件

  indexes {
    (獎助單位帳號, 獎學金編號) [pk]    // 複合主鍵
  }
}

// 推薦信（弱實體，依賴 老師 + 申請資料）
Table 推薦信 {
  老師帳號 varchar   [not null, ref: > 老師.帳號]     // FK → 老師.帳號（填寫）
  申請編號 integer   [not null, ref: > 申請資料.申請編號] // FK → 申請資料.申請編號（附上）
  推薦信編號 integer [not null]                     // 部分鍵：推薦信編號

  推薦內容 text          // 推薦內容
  填寫日期 date          // 填寫日期

  indexes {
    (老師帳號, 申請編號, 推薦信編號) [pk]
  }
}

// 成績（學生的歷年成績紀錄）
Table 成績 {
  學生帳號 varchar [not null, ref: > 學生.帳號] // FK → 學生.帳號
  學年 varchar   [not null]                  // 學年
  學期 varchar   [not null]                  // 學期

  平均成績 numeric      // 平均成績
  GPA numeric           // GPA
  班排 integer          // 班排
  全班人數 integer      // 全班人數

  indexes {
    (學生帳號, 學年, 學期) [pk]
  }
}

// 申請成績（紀錄「某筆申請用了哪幾筆成績」）
Table 申請成績 {
  申請編號 integer  [not null, ref: > 申請資料.申請編號] // FK → 申請資料.申請編號
  學生帳號 varchar [not null, ref: > 學生.帳號]       // FK → 學生.帳號
  學年 varchar    [not null]
  學期 varchar    [not null]

  indexes {
    (申請編號, 學生帳號, 學年, 學期) [pk]
  }
}

// 審查紀錄（公告）
Table 審查紀錄 {
  審查紀錄編號 integer [primary key]           // 審查紀錄編號
  審查日期 date                               // 審查日期
  審查結果 varchar                            // 審查結果
  備註 text                                   // 備註
  系統管理員帳號 varchar [not null, ref: > 系統管理員.帳號]  // FK → 系統管理員.帳號
}



// =============  關聯 (Ref) 定義（只保留「複合 FK」）  =============

// ★ 獎(助)學金 – 申請資料（屬於）：複合外鍵
Ref: 獎助學金.(獎助單位帳號, 獎學金編號)
  < 申請資料.(獎助單位帳號, 獎學金編號)

// ★ 成績 – 申請成績（寫入）：複合外鍵
Ref: 成績.(學生帳號, 學年, 學期)
  < 申請成績.(學生帳號, 學年, 學期)
